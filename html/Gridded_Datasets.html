Site level NH3 concentration estimates

The ste_nh3_concs() function returns a data.frame or data.table or site level NH3 concentration estimates, where the mean_nh3 concentration represents an area weighted mean.

Return area weighted NH3 concentrations for selected SACs

  • Argument yrs defaults to 2019 - 2021
  • Setting agr_yrs = T returns the average across the 3 years (DEFAULT)
  • A data.table is returned as default, setting as_dt = F returns a data.frame
  • Lowering prcsn to improves accuracy (and increases processing time)
  • The column DIFF_HA in output indicated difference in area of sf object to raster representation
dt <- ste_nh3_concs(ste_type = "SAC",id = c("UK0030322","UK0016599"),as_dt = T, prcsn = 1)

Plotting NH3 concentrations

Using ggplot

# lazy load NH3 concs
r_nh3 <- conc_nh3_r(yrs = 2019:2021)
names(r_nh3) <- 2019:2021

# Set colour breaks
brks <- c(0,1,2,3,4)
# Set colour ramp
cols <- c("#85c6ff",brewer.pal(length(brks)-1,"YlOrRd"))

# plot concs
SpatialUKCEH::plot_r_cats(legnd_txt="NH[3]~'Concentrations'", 
                  unit = "'('*mu*g~NH[3]~m^-3*')'",
                  r = r_nh3,
                  cols = cols,
                  ncols = 3,
                  brks = brks)
## <SpatRaster> resampled to 500760 cells.

Using leaflet

# Return calibrated EMEP4UK ammonia concentrations (2019 - 2021)
# aggregate to 3 year mean
r_nh3_conc_19_21 <- conc_nh3_r(yrs = 2019:2021) %>% app(., mean, na.rm = T)
crs(r_nh3_conc_19_21) <- "EPSG:27700"

## project to 3857
r_conc_19_21_3857 <- project(r_nh3_conc_19_21, "EPSG:3857", res = 1000) 

cols <- c("#9ce4ff","#fbff87","#ffa754","#ff5754","#b36bff")
brks <- c(0,1,2,3,4,99)
labs <- c("0 - 1","> 1 - 2","> 2 - 3","> 3 - 4","> 4")
pal <- colorBin(palette = cols,
                   bins = brks,
                   domain = values(r_conc_19_21_3857),
                na.color = "transparent")

leaflet() %>%
addProviderTiles(providers$Esri.WorldGrayCanvas) %>%
addRasterImage(x = r_conc_19_21_3857, colors = pal, project = F) %>% 
addLegend(colors = cols, 
          labels = labs, values = values(r_conc_19_21_3857))
